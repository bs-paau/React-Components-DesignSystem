## Description:
Card Component with different properties.

## Installation:
```
$ npm install @*project*/card@latest
```

```
import {Card} from '@*project*/card';
```


### Example:
```js
import React from 'react';

import {Card} from '@*project*/card';


const CardContainer = () => {

    return (
        <div>
            <Card type='greyDark'>
            </Card>

            <Card type='white'>
            </Card>

            <Card type='orangeStandard'>
            </Card>
        </div>
    );
};
export default CardContainer;
```
